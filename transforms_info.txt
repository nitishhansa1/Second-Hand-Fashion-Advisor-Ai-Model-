if IN_COLAB:
    # Colab environment paths
    DATASET_ROOT = "/content/fashion_data"
    ARTIFACTS_ROOT = "/content/drive/MyDrive/fashion_ai"

    DATASET_ROOT = "D:/fashion-api/archive/myntradataset/images_processed" # Example local path for processed images

# --- Defensive check for DATASET_ROOT ---
if not os.path.exists(DATASET_ROOT):
    raise FileNotFoundError(f"Dataset root directory '{DATASET_ROOT}' does not exist. Please ensure the data preparation cell (e.g., b6152b2e) has been run successfully to create and populate it.")
if not any(os.path.isdir(os.path.join(DATASET_ROOT, d)) for d in os.listdir(DATASET_ROOT)):
    raise FileNotFoundError(f"Dataset root directory '{DATASET_ROOT}' is empty or contains no class subdirectories. Please ensure the data preparation cell (e.g., b6152b2e) has populated the directory with category folders.")

# 1. Define Transforms: training and validation
# Note: Normalization with ImageNet stats is included in both transforms as it's critical for ResNet performance
# and was present in the original setup, despite the user's initial instruction to omit it for aggressive augmentation.

# Train Transform: Aggressive augmentation
train_transform = transforms.Compose([
    transforms.Resize((256, 256)),
    transforms.RandomCrop(224),
    transforms.RandomHorizontalFlip(p=0.5),
    transforms.ColorJitter(brightness=0.3, contrast=0.3, saturation=0.3, hue=0.1),
    transforms.RandomRotation(degrees=15),
    transforms.ToTensor(),
    transforms.Normalize([0.485, 0.456, 0.406], [0.229, 0.224, 0.225]) # CRITICAL: ImageNet stats for ResNet
])

# Validation Transform: Minimal augmentation
val_transform = transforms.Compose([
    transforms.Resize((224, 224)),
    transforms.ToTensor(),
    transforms.Normalize([0.485, 0.456, 0.406], [0.229, 0.224, 0.225]) # CRITICAL: ImageNet stats for ResNet
])


# 2. Load the full dataset (initially with a placeholder transform to get counts/classes)
full_dataset = datasets.ImageFolder(DATASET_ROOT, transform=val_transform) # Using val_transform initially

# 3. Split dataset into training and validation sets
# Determine sizes for train and validation (e.g., 80% train, 20% val)
train_size = int(0.8 * len(full_dataset))
val_size = len(full_dataset) - train_size
train_dataset, val_dataset = random_split(full_dataset, [train_size, val_size])

# Apply specific transforms to the subsets
# NOTE: This approach modifies the underlying dataset transform. For a clean split, you would typically
# create two separate ImageFolder instances with different transforms. This works for random_split.
train_dataset.dataset.transform = train_transform
val_dataset.dataset.transform = val_transform

print(f"Total dataset size: {len(full_dataset)}")
print(f"Training set size: {len(train_dataset)}")
print(f"Validation set size: {len(val_dataset)}")


# --- Fix Class Imbalance using WeightedSampler (for training set only) ---
# Need to reconstruct class counts based on the actual classes in train_dataset
# The original full_dataset has labels, but random_split shuffles indices.
# We need to map the indices of train_dataset back to the original full_dataset's labels.
train_labels = [full_dataset.targets[idx] for idx in train_dataset.indices]
class_counts = [0] * len(full_dataset.classes)
for label in train_labels:
    class_counts[label] += 1

weights = [1.0 / count if count > 0 else 0 for count in class_counts]
sample_weights = [weights[label] for label in train_labels]

train_sampler = WeightedRandomSampler(
    weights=sample_weights,
    num_samples=len(train_dataset),
    replacement=True
)

# Use the sampler for the training loader
train_loader = DataLoader(train_dataset, batch_size=32, sampler=train_sampler)
val_loader = DataLoader(val_dataset, batch_size=32, shuffle=False) # No need for sampler for validation

# Store classes for later use (e.g., in `c1da3308`)
classes = full_dataset.classes
